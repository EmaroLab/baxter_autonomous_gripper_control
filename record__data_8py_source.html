96  '''
97  To maintain the time shift and the differences list, two variables are saved each time a pack arrives:
98  - last_local_time : ros timestamp of the last pack
99  - last_pack_time : smartwatch timestamp of the last pack
100 
101  First pack will be thrown away
102  '''
103  if (is_first_pack):
104  last_local_time = round(rospy.get_rostime().to_nsec()*NANOS_TO_MILLIS)
105  last_pack_time = max(data.angular_velocity[-1].time.data, data.linear_acceleration[-1].time.data)
106  pack_time_list.append(last_pack_time)
107  is_first_pack = False
108  return
109 
110  packCounter+=1
111 
112  #saving list of Vector3Time adding what remains from the previous packet
113  av = av_remains_before + data.angular_velocity
114  la = la_remains_before + data.linear_acceleration
115 
116  packlen = manage_remains(av,la)
117 
118  #time shift calculation with a 'diff' list called 'shift'
119  for i in range (0, packlen):
120  pack_time_list.append(max(av[i].time.data, la[i].time.data))
121  shift = [t - s for s, t in zip(pack_time_list, pack_time_list[1:])]
122 
123  watch_rows[0] = []
124  for i in range (0, packlen):
125  t_stamp = last_local_time + sum(shift[0:i+1])
126  row=[packCounter, av[i].vector.x, av[i].vector.y, av[i].vector.z, \
127  la[i].vector.x, la[i].vector.y, la[i].vector.z, t_stamp]
128  watch_rows[0].append(row)
129  watch_rows[1]=1

132  last_pack_time = max(av[packlen-1].time.data, la[packlen-1].time.data)
